# 安装miniconda 网址为 https://docs.anaconda.com/miniconda/miniconda-other-installer-links/

# 创建虚拟环境
conda create -n lrc python=3.10
# 激活虚拟环境
conda activate lrc
# 安装各种模块
conda install pyserial
conda install numpy
conda install pandas
conda install psutil
conda install attrs
conda install pyqtgraph
pip install PyQt5
pip install matplotlib
pip install flask -i https://mirrors.tuna.tsinghua.edu.cn/pypi/web/simple/
pip install flask-cors
pip install requests

# 安装MQTT中间件
sudo apt-get install mosquitto
# 安装MQTT客户端
pip install paho-mqtt==1.6.1

# 如果需要远程vnc运行交互程序
sudo apt-get update
sudo apt-get install xorg libxcb-xinerama0
